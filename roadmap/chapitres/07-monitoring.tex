\chapter{Monitoring \& observabilité}
\label{chap:monitoring}

L'observabilité repose sur \textbf{Prometheus + Grafana} (métriques),
\textbf{ELK ou Loki} (logs), \textbf{AlertManager} (alertes) et
\textbf{Jaeger} (tracing distribué).

\section{Règles d'alerte}

\begin{center}
\small
\begin{tabular}{lp{5.6cm}lp{4.4cm}}
\toprule
\textbf{Alerte} & \textbf{Condition} & \textbf{Sévérité} & \textbf{Action} \\
\midrule
Latence API élevée &
  p95 sur 5 min $> 500$ ms &
  \textcolor{miel}{\textbf{Warning}} &
  Notifier Slack + escalade si persistant \\
\addlinespace
Taux d'erreurs $> 1\,\%$ &
  \texttt{http\_errors} / \texttt{http\_requests} $> 0{,}01$ &
  \textcolor{red!70!black}{\textbf{Critical}} &
  PagerDuty + rollback automatique \\
\addlinespace
Connexions DB épuisées &
  actives / max $> 0{,}9$ &
  \textcolor{red!70!black}{\textbf{Critical}} &
  Scaler le pool + alerte ops \\
\addlinespace
JVM heap $> 85\,\%$ &
  heap utilisé / heap max $> 0{,}85$ &
  \textcolor{miel}{\textbf{Warning}} &
  Analyse mémoire + redémarrage planifié \\
\bottomrule
\end{tabular}
\end{center}

\section{Gestion des logs}

\begin{itemize}
  \item \textbf{Niveaux} : ERROR, WARN, INFO, DEBUG ;
  \item \textbf{Corrélation} : en-tête \texttt{X-Request-ID} propagé pour le
        tracing distribué ;
  \item \textbf{Rétention} : 30 jours en stockage chaud, 1 an en stockage
        froid (S3).
\end{itemize}

\begin{encadre}
\textbf{Boucle DevOps.} Les métriques métier (production, visites, alertes
sanitaires) sont exposées par l'application elle-même : les dashboards de
supervision technique et les tableaux de bord fonctionnels du cahier des
charges (\S4.3) partagent ainsi la même source de vérité.
\end{encadre}
