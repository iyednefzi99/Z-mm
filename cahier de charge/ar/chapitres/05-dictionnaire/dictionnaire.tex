\section{اصطلاحات الأنواع}
الأنواع المستعملة هي: عدد صحيح، عدد عشري، نص، تاريخ، ساعة، تاريخ ووقت، منطقي،
تعداد، ومرجع (مفتاح خارجي نحو كيان آخر). تُقرَن المقادير الفيزيائية بوحدة قابلة
للضبط لإتاحة تدويل أنظمة القياس.

ترجمتُه العلائقية المُطبَّعة (المفاتيح وأنواع \textenglish{SQL} وقيود

(الشكل~\ref{fig:mld}). يُعطى تقابل الأسماء بين القاموس ومخطّط الأصناف والنموذج
المنطقي في الجدول~\ref{tab:vocabulaire}.
\end{encadre}

\section{النتائج وقواعد الحساب (المُخرجات)}
\begin{itemize}
  \item \textbf{كمية عسل خلية}: مجموع أوزان إطارات العسل في العاسلات، مطروحًا منه وزن الفارغ.
\end{itemize}
\[ \text{HoneyQuantity}(h) = \sum_{f \in \text{honeyFrames}(h)} \text{weight}(f) - \text{tare}(h) \]
تُعرَض هذه القيمة عبر خدمة الويب من خلال \texttt{getZummHoneyActualQuantity(zummID)}.
\begin{itemize}
  \item \textbf{العائد على الاستثمار} لموقع أو خلية:
\end{itemize}
\[ \text{ROI} = \frac{\text{Valued production}}{\text{Operation costs}} \]
\begin{itemize}
  \item \textbf{تنبيه الجمع}: يُطلَق إذا تجاوز الإنتاج العتبة القابلة للضبط.
\end{itemize}
\[ \text{Production}(h) > \text{ProductionThreshold} \Rightarrow \text{harvest or extension alert} \]
\begin{itemize}
  \item \textbf{تنبيه الانخفاض}: يُطلَق إذا تجاوز الانخفاض النسبي العتبة.
\end{itemize}
\[ \frac{V_{\text{prev}} - V_{\text{current}}}{V_{\text{prev}}} > \text{DropThreshold} \Rightarrow \text{inspection alert} \]
\begin{itemize}
  \item \textbf{تحويل الوحدات} نحو الوحدة المرجعية، لمقارنة قياسات غير متجانسة:
\end{itemize}
\[ V_{\text{reference}} = V_{\text{measurement}} \times \text{conversionFactor}(\text{unit}) \]
