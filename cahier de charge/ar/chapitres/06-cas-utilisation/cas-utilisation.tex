\chapter{حالات الاستعمال المفصّلة}
%==============================================================================
يصف هذا الجزء كيف يُنجز النظام عمليات المستخدمين. الفاعلون هم مربّي النحل وعون
تربية النحل والعون المشرف والمسؤول ومدير النظام والتطبيقات الخارجية.

يعرض مخطّط حالات الاستعمال الشامل (الشكل \ref{fig:uc}) جميع الفاعلين، وتعميم
العون المشرف انطلاقًا من عون تربية النحل، إضافةً إلى كل التبعيات بين الحالات
(علاقتا التضمين والتوسيع).

\section{UC1: تخطيط زيارة والمصادقة عليها}
\begin{itemize}
  \item \textbf{الفاعل}: العون المشرف.
  \item \textbf{الشروط المسبقة}: الخلية موجودة والعون مُوثَّق.
  \item \textbf{السيناريو الاسمي}: يقترح العون تاريخ زيارة، يراجع المشرف البرنامج ثم يصادق، ويسجّل النظام الحالة المُصادَق عليها.
  \item \textbf{البدائل}: يرفض المشرف ويطلب تاريخًا جديدًا.
  \item \textbf{الشرط اللاحق}: الزيارة مبرمَجة وظاهرة في الرزنامة.
\end{itemize}

\section{UC2: إنجاز زيارة وملء التقرير}
\begin{itemize}
  \item \textbf{الفاعل}: عون تربية النحل المسؤول عن الخلية.
  \item \textbf{الشروط المسبقة}: توجد زيارة مبرمَجة ومُصادَق عليها.
  \item \textbf{السيناريو الاسمي}: يُدخل العون التاريخ والساعة والمدة والسبب والملاحظات والإجراءات المقرّرة والمنجَزة والتوصيات، ثم يقيّم العدد والحالة الصحية والإنتاجية من 1 إلى 3.
  \item \textbf{البدائل}: إدخال دون اتصال ثم مزامنة مؤجّلة، وإضافة صور.
  \item \textbf{الشرط اللاحق}: التقرير محفوظ وتنتقل خانة الرزنامة إلى حالة «مُنجَز».
\end{itemize}

\section{UC3: استعراض لوحة قيادة الإنتاج}
\begin{itemize}
  \item \textbf{الفاعل}: مربّي النحل أو المالك.
  \item \textbf{الشروط المسبقة}: توجد قياسات إنتاج.
  \item \textbf{السيناريو الاسمي}: يصفّي مربّي النحل حسب المزرعة والموقع والفترة، ويعرض النظام الخلايا التي يتجاوز إنتاجها العتبة ويقترح الجمع أو التمديد.
  \item \textbf{الشرط اللاحق}: يقرّر مربّي النحل تدخّلًا.
\end{itemize}

\section{UC4: معالجة تنبيه صحي}
\begin{itemize}
  \item \textbf{الفاعل}: المسؤول.
  \item \textbf{الشروط المسبقة}: تمّ الكشف عن انخفاض غير طبيعي.
  \item \textbf{السيناريو الاسمي}: يُشير النظام إلى الخلية المعنية، ويبرمج المسؤول تفتيشًا وشيكًا.
  \item \textbf{الشرط اللاحق}: زيارة تفتيش مبرمَجة.
\end{itemize}

\section{UC5: الاستعلام عن كمية العسل عبر الواجهة البرمجية}
\begin{itemize}
  \item \textbf{الفاعل}: تطبيق خارجي (\textenglish{Excel} أو غيره).
  \item \textbf{الشروط المسبقة}: للتطبيق وصول آمن إلى خدمة الويب.
  \item \textbf{السيناريو الاسمي}: يستدعي التطبيق \texttt{getZummHoneyActualQuantity(zummID)}، وتحسب الخدمة كمية العسل وتُرجعها.
  \item \textbf{الشرط اللاحق}: تُعاد القيمة بصيغة خدمة الويب \textenglish{XML}.
\end{itemize}
