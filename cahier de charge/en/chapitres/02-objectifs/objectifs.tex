%==============================================================================
\chapter{Project objectives}
%==============================================================================

\section{Main objective}
Provide a \textbf{multi-tier, scalable and loosely coupled} application that
manages the entire life cycle of hives, from their physical composition to
production monitoring, and offers decision-support dashboards to the various
user profiles.

\section{Specific objectives}
\begin{itemize}
  \item Ensure \textbf{CRUD} operations (create, read, update, delete) on all business entities.
  \item Manage the \emph{farmer / farm / site / hive / body or super / frame} hierarchy.
  \item Plan, approve and track the inspection visits of beekeeping agents.
  \item Produce structured and actionable \textbf{visit reports}.
  \item Offer filterable \textbf{dashboards} (agents $\times$ hives calendar, production alerts, health alerts, financial summary).
  \item Anticipate the integration of \textbf{time-stamped and geolocated} measurements from sensors with heterogeneous units.
  \item Guarantee \textbf{internationalisation}, \textbf{security} and \textbf{interoperability} (web services).
\end{itemize}

\section{Expected results (success indicators)}
\begin{itemize}
  \item Support for at least \textbf{three languages} (FR, EN, AR) in the demonstration version.
  \item Automatic detection of \textbf{configurable} threshold breaches (production, population).
  \item Presentation of production data by farm, site and hive as charts.
  \item Exposure of an API that can be queried by third-party applications (Excel, etc.).
\end{itemize}
